
to create databse : 
    CREATE DATABSE ' database name ' 

to set default schema 
    USE 'database name' 

to drop a database 
    DROP DATABASE 'databse name'

 to set our databse to read only 
    ALTER DATABASE ' myDB ' READ ONLY = 1;

to undo read only mode 
    ALTER DATABSE 'name of database ' READ ONLY = 0;
